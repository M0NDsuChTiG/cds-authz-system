Quick Start

bash — install.sh 
# 1. Install infrastructure
git clone https://github.com/M0NDsuChTiG/cds-authz-system.git
cd cds-authz-system && ./install.sh

# 2. Import trust root
cds-cli key import --id prod --path ./public.pub

# 3. Authorize image digest
cds-cli trust add alpine:latest --key-id prod --digest sha256:25109...
